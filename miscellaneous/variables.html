Default value : (): KeyMap => { const listMap: KeyMap = {}; listMap.Enter = splitListItem(schema.nodes.list_item); listMap['Mod-['] = liftListItem(schema.nodes.list_item); listMap['Mod-]'] = sinkListItem(schema.nodes.list_item); listMap.Tab = sinkListItem(schema.nodes.list_item); return listMap; }
getPlugins
Default value : (options: ComputedOptions): Plugin[] => { const historyKeyMap = getHistoryKeyMap(); const listKeyMap = getListKeyMap(); const plugins = [ history(), keymap({ 'Mod-b': toggleMark(schema.marks.strong), 'Mod-i': toggleMark(schema.marks.em), 'Mod-`': toggleMark(schema.marks.code), }), keymap(historyKeyMap), keymap(listKeyMap), keymap(baseKeymap), placeholder(options.placeholder), ]; if (options.toolbar) { plugins.push(menu(options.toolbar)); } return plugins; }
